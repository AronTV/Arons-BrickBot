async function loadProduct(id) { const res = await fetch(`/api/search/${id}`); const json = await res.json(); const info = document.getElementById("info"); const title = document.getElementById("title"); const image = document.getElementById("image"); const priceSetdb = document.getElementById("price-setdb"); const priceBluebrixx = document.getElementById("price-bluebrixx"); const status = document.getElementById("status"); const input = document.getElementById("articleInput"); if (!json.success) { info.innerHTML = "❌ Produkt nicht gefunden"; return; } const data = json.data; /* ========================= BASIC DATA ========================= */ title.innerText = data.name ?? "-"; image.src = data.image ? `/api/image?url=${encodeURIComponent(data.image)}` : ""; priceSetdb.innerText = data.setdb?.price ? `${data.setdb.price} €` : "-"; priceBluebrixx.innerText = data.bluebrixx?.price ? `${data.bluebrixx.price} €` : "-"; status.innerText = data.bluebrixx?.status ?? "-"; /* ========================= EXTRA INFO ========================= */ const extraInfo = `
EAN: ${data.ean ?? "-"}
Teileanzahl: ${data.parts ?? "-"}
Minifiguren: ${data.minifigures ?? "-"}
Altersempfehlung: ${data.age ?? "-"}
Maße: ${data.dimensions ?? "-"}
Gewicht: ${data.weight ?? "-"}
Erscheinungsjahr: ${data.year ?? "-"}
Themenwelt/Serie: ${data.theme ?? "-"}
Bewertung: ${data.rating ?? "-"}
Preis/Stein: ${data.pricePerPart ?? "-"}
`; /* ========================= BUTTONS ========================= */ const buttons = `
${data.setdb?.url ? ` 📚 In SetDB öffnen ` : ""} ${data.bluebrixx?.url ? ` 🛒 Bei BlueBrixx öffnen ` : ""}
`; /* ========================= RENDER ========================= */ info.innerHTML = extraInfo + buttons; /* ========================= INPUT CLEANUP (BONUS) ========================= */ if (input) { input.value = ""; } } /* ========================= ENTER KEY SUPPORT (BONUS FIX) ========================= */ document.addEventListener("DOMContentLoaded", () => { const input = document.getElementById("articleInput"); if (!input) return; input.addEventListener("keydown", (e) => { if (e.key === "Enter") { loadProduct(input.value); } }); });
